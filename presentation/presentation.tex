\begin{frame}{What is Handwriting Recognition?}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item Input: sequence of $(x, y)$ coordinates captured by a stylus
        \item Goal: identify the drawn character (letter or digit)
        \item Origin: GRAIL system (RAND Corporation, 1966) — real-time recognition on an IBM System/360
      \end{itemize}
    \end{column}
    \begin{column}{0.40\textwidth}

      \end{tikzpicture}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Why Is This Interesting?}
  \begin{block}{The key challenge}
    \begin{itemize}
      \item tablet discretisation introduces jaggies in every stroke
      \item A single character produces dozens of noisy, redundant coordinate
    points
    \item two people writing the same letter look very different
    on paper
    \item two diffrent characters can have many similarities
    \item ideally we can recognize the character fast
    \end{itemize}
  \end{block}

  \bigskip
  \begin{example}[the digit \textbf{5} vs the letter \textbf{S}]
    \begin{itemize}
      \item Drawn in one stroke, left $\to$ down $\to$ right $\to$ down $\to$ left
      \item Corners resolve the ambiguity
    \end{itemize}
  \end{example}
\end{frame}

\section{Solution: The Pipeline}
% ============================================================

\begin{frame}{Approache to Character Recognition}
  \textbf{Feature-based (Groner)}
  \begin{itemize}
    \item Extract compact stroke features
    \item Rule-based decision tree
    \item Fast, interpretable, works in real-time
  \end{itemize}

  \bigskip
  \begin{alertblock}{Why Groner's pipeline suits SQL}
    Each processing stage is a \emph{set transformation} on an ordered
    sequence of points — a natural fit for relational queries with
    window functions and recursive CTEs.
  \end{alertblock}
\end{frame}

\begin{frame}{Groner's Five-Stage Pipeline}
  \begin{center}

    \begin{tikzpicture}[

      \node[box]                    (raw)    {Raw\\Points};
      \node[box, right=of raw]      (smooth) {Smoothing};
      \node[box, right=of smooth]   (thin)   {Thinning};
      \node[box, right=of thin]     (curv)   {Curvature};
      \node[box, right=of curv]     (corn)   {Corner\\Detection};
      \node[box, right=of corn]     (dec)    {Decision\\Tree};

      \draw[arr] (raw)    -- (smooth);

  \end{center}

  \alert{Hier ggf. an jeden Pipeline Schritt notieren, welche Herausforderung gefixt wird.}


  \begin{itemize}
    \item Each stage is one SQL \texttt{WITH} clause — the full pipeline is a single query
    \item Running example: digit \textbf{3} passes through each stage
  \end{itemize}
\end{frame}


% ── Stage 1 ─────────────────────────────────────────────────
\begin{frame}[fragile]{Stage 1 — Smoothing}
  Remove quantisation noise with an exponential moving average
  (IIR low-pass filter): each smoothed point is a weighted blend
  of the previous smoothed point and the new raw point.
  \[
    x^s_i = \alpha \cdot x^s_{i-1} + (1-\alpha) \cdot x^r_i
    \qquad
    y^s_i = \alpha \cdot y^s_{i-1} + (1-\alpha) \cdot y^r_i
  \]

  \alert{Here a example + visualization like in the blog article!}
\end{frame}

\begin{frame}[fragile]{Stage 1 — Smoothing Implementation}
  \begin{lstlisting}[basicstyle=\ttfamily\footnotesize]
SELECT pos,
        x,
        y
    FROM raw_data
    WHERE pos = 1
    UNION ALL
    SELECT r.pos,
        get('factor')*s.x + (1 - get('factor')) * r.x AS sx,
        get('factor')*s.y + (1 - get('factor')) * r.y AS sy
    FROM raw_data r
        JOIN smoothing_phase s ON r.pos = s.pos + 1
  \end{lstlisting}
\end{frame}

% ── Stage 2 ─────────────────────────────────────────────────
\begin{frame}[fragile]{Stage 2 — Thinning}
  Discard points too close to the last accepted point $(x^t_{j-1}, y^t_{j-1})$
  to remove redundancy from slow pen movement:
  \[
    \text{accept } (x^s_i, y^s_i)
    \iff
    \sqrt{(x^s_i - x^t_{j-1})^2 + (y^s_i - y^t_{j-1})^2} \;>\; \tau
  \]
  \begin{itemize}
    \item Threshold $\tau$ defines an acceptance radius around the last kept point
    \item Original GRAIL used a bounding-square test ($|\Delta x| \geq \tau$ or $|\Delta y| \geq \tau$);
          we use Euclidean distance — rotation-invariant
    \item Pipeline value: $\tau = 0.2$ (normalised coordinates)
  \end{itemize}

  \alert{Here a example + visualization like in the blog article!}
\end{frame}

\begin{frame}[fragile]{Stage 2 — Thinning Implementation}
  \begin{lstlisting}[basicstyle=\ttfamily\footnotesize]
    SELECT pos, x, y
    FROM smoothing_phase
    WHERE pos = 1
    UNION ALL
    SELECT next_keep.pos, next_keep.x, next_keep.y
    FROM thinning_scan ts
    CROSS JOIN LATERAL (
        SELECT pos, x, y
        FROM smoothing_phase
        WHERE pos > ts.pos
          AND sqrt((x - ts.x)^2 + (y - ts.y)^2) > getvariable('thinning_threshold')
        ORDER BY pos
        LIMIT 1
    ) next_keep
  \end{lstlisting}
\end{frame}

% ── Stage 3 ─────────────────────────────────────────────────
\begin{frame}[fragile]{Stage 3 — Curvature Detection}
  Assign one of four cardinal directions to each segment by comparing
  the bounding rectangle's width to its height
  ($\Delta x = x^t_i - x^t_{i-1}$, $\Delta y = y^t_i - y^t_{i-1}$):
  \[
    d_i =
    \begin{cases}
      \text{right} / \text{left} & \text{if } |\Delta x| \geq |\Delta y|,
                                   \text{ sign of } \Delta x \\[2pt]
      \text{up} / \text{down}    & \text{if } |\Delta x| < |\Delta y|,
                                   \text{ sign of } \Delta y
    \end{cases}
  \]
  Only \emph{changes} in direction are kept — e.g.\ \textbf{L} becomes
  $\downarrow\,\rightarrow$, \textbf{O} becomes
  $\leftarrow\,\downarrow\,\rightarrow\,\uparrow$.

  \alert{Here a example + visualization like in the blog article!}
\end{frame}

\begin{frame}[fragile]{Stage 3 — Curvature Detection Implementation}
  \begin{lstlisting}[basicstyle=\ttfamily\footnotesize]
SELECT pos, x, y,
    CASE
        WHEN LAG(x) OVER w IS NULL THEN NULL
        WHEN ABS(x - LAG(x) OVER w) >= ABS(y - LAG(y) OVER w)
        THEN CASE WHEN x >= LAG(x) OVER w THEN 'right' ELSE 'left' END
        ELSE CASE WHEN y >= LAG(y) OVER w THEN 'up' ELSE 'down' END
    END AS direction
FROM thinning_scan
WINDOW w AS (ORDER BY pos)
  \end{lstlisting}
\end{frame}

% ── Stage 4 ─────────────────────────────────────────────────
\begin{frame}[fragile]{Stage 4 — Corner Detection}
  Corners need finer angular resolution — encode each segment as one of
  16 sectors via $\theta_i = \operatorname{atan2}(\Delta y, \Delta x)$:
  \[
    d_i = \left\lfloor
      \frac{(\theta_i + 11.25^\circ) \bmod 360^\circ}{22.5^\circ}
    \right\rfloor \in \{0, \dots, 15\}
  \]
  A corner is flagged at point $i$ when all three hold:
  \begin{itemize}
    \item Two segments before share a direction: $d_{i-1} = d_{i-2}$
    \item Two segments after share a direction: $d_{i+1} = d_{i+2}$
    \item Angular change $\geq 90^\circ$ (4 of 16 sectors), with wrap-around:
      \[
        \min\!\bigl(|d_{i+1} - d_{i-1}|,\; 16 - |d_{i+1} - d_{i-1}|\bigr) \geq 4
      \]
  \end{itemize}

  \alert{Here a example + visualization like in the blog article!}
\end{frame}

\begin{frame}[fragile]{Stage 4 — Corner Detection Implementation}
  \begin{lstlisting}[basicstyle=\ttfamily\footnotesize]
SELECT pos, x, y
FROM directions_16_calc
WHERE direction IS NOT NULL
WINDOW w3 AS (ORDER BY pos)
QUALIFY
    LEAST(ABS(LEAD(direction, 1) OVER w3 - LAG(direction, 1) OVER w3),
          16 - ABS(LEAD(direction, 1) OVER w3 - LAG(direction, 1) OVER w3)) >= 4
    AND LAG(direction, 1) OVER w3 = LAG(direction, 2) OVER w3
    AND LEAD(direction, 1) OVER w3 = LEAD(direction, 2) OVER w3
  \end{lstlisting}
\end{frame}

% ── Stage 5 ─────────────────────────────────────────────────
\begin{frame}[fragile]{Stage 5 — Character Recognition}
  Combine direction sequence, corner count/position, start/end cell,
  and aspect ratio to identify the character:

  \begin{lstlisting}[basicstyle=\ttfamily\footnotesize]
character_recognition AS (
    SELECT DISTINCT cdr.char
    FROM first_directions fd
    CROSS JOIN character_direction_rules cdr
    WHERE
        fd.dirs[1:len(cdr.prefix)] = cdr.prefix
)
  \end{lstlisting}
  \alert{Noch nicht vollständig!}
\end{frame}

% ── Full pipeline ────────────────────────────────────────────
\begin{frame}[fragile]{Putting It All Together}
  The complete recogniser is \emph{one SQL query}:

  \begin{lstlisting}[basicstyle=\ttfamily\footnotesize]
WITH
  smoothed   AS ( ... ),   -- Stage 1: window avg
  thinned    AS ( ... ),   -- Stage 2: recursive CTE
  directions AS ( ... ),   -- Stage 3: CASE + LAG
  corners    AS ( ... ),   -- Stage 4: direction diff
  result     AS ( ... )    -- Stage 5: CASE cascade
SELECT character FROM result;
  \end{lstlisting}

\end{frame}

\section{Conclusion}

\begin{frame}{Limitations \& Outlook}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Limitations}
      \begin{itemize}
        \item Single-stroke characters only\\(Groner's original: multi-stroke)
        \item Decision tree hand-coded\\— biased toward one writing style
        \item Recursive thinning: row-by-row,\\not easily parallelised
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Outlook}
      \begin{itemize}
        \item Extend to multi-stroke via\\pen-up/pen-down events
        \item Learn thresholds from data\\(SQL + ML hybrid)
        \item Compare with modern DNN\\approaches (trade-offs)
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Summary}
  \begin{itemize}
    \item \textbf{Problem:} real-time recognition of handprinted characters
          from noisy coordinate streams
    \item \textbf{Solution:} Groner's 5-stage feature-extraction pipeline,
          implemented as a single SQL \texttt{WITH} query
    \item \textbf{Key SQL constructs:} window functions (smoothing, direction),
          recursive CTE (thinning), \texttt{CASE} cascade (decision tree)
  \end{itemize}

\end{frame}
